import "jsr:@supabase/functions-js/edge-runtime.d.ts"; import { createClient } from "npm:@supabase/supabase-js@2"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type, x-admin-secret", "Access-Control-Allow-Methods": "POST, OPTIONS", }; Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: corsHeaders }); try { const adminSecret = req.headers.get("x-admin-secret"); if (adminSecret !== Deno.env.get("ARTWORK_ADMIN_SECRET")) { return json({ success: false, error: "Unauthorized" }, 401); } const { mistake_id, submission_id } = await req.json(); if (!mistake_id && !submission_id) { return json({ success: false, error: "Missing mistake_id or submission_id" }, 400); } const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); const openaiKey = Deno.env.get("OPENAI_API_KEY"); if (!openaiKey) return json({ success: false, error: "Missing OPENAI_API_KEY" }, 500); let item: any; let mode: "mistake" | "submission"; if (submission_id) { mode = "submission"; const { data, error } = await supabase .from("mistake_submissions") .select("*") .eq("id", submission_id) .single(); if (error || !data) return json({ success: false, error: "Submission not found", details: error }, 404); item = data; } else { mode = "mistake"; const { data, error } = await supabase .from("mistakes") .select("*") .eq("id", mistake_id) .single(); if (error || !data) return json({ success: false, error: "Mistake not found", details: error }, 404); item = data; } const prompt = buildArtworkPrompt(item); const imageRes = await fetch("https://api.openai.com/v1/images/generations", { method: "POST", headers: { Authorization: `Bearer ${openaiKey}`, "Content-Type": "application/json", }, body: JSON.stringify({ model: "gpt-image-1", prompt, size: "1024x1024", quality: "medium", n: 1, }), }); const imageJson = await imageRes.json(); if (!imageRes.ok) { return json({ success: false, error: "OpenAI failed", details: imageJson }, 500); } const b64 = imageJson?.data?.[0]?.b64_json; if (!b64) return json({ success: false, error: "No image returned", details: imageJson }, 500); const bytes = Uint8Array.from(atob(b64), c => c.charCodeAt(0)); const idForFile = submission_id || mistake_id; const fileName = `${mode}-${idForFile}-${Date.now()}.png`; const { error: uploadError } = await supabase.storage .from("mistake-artwork") .upload(fileName, bytes, { contentType: "image/png", upsert: true }); if (uploadError) return json({ success: false, error: "Upload failed", details: uploadError }, 500); const { data: urlData } = supabase.storage .from("mistake-artwork") .getPublicUrl(fileName); const artwork_url = urlData.publicUrl; if (mode === "submission") { const { error: updateError } = await supabase .from("mistake_submissions") .update({ ai_artwork_url: artwork_url, ai_artwork_prompt: prompt, ai_artwork_status: "generated", ai_artwork_generated_at: new Date().toISOString(), }) .eq("id", submission_id); if (updateError) return json({ success: false, error: "Submission artwork update failed", details: updateError }, 500); } else { const { error: updateError } = await supabase .from("mistakes") .update({ artwork_url, artwork_prompt: prompt, artwork_status: "generated", artwork_generated_at: new Date().toISOString(), }) .eq("id", mistake_id); if (updateError) return json({ success: false, error: "Mistake artwork update failed", details: updateError }, 500); } return json({ success: true, mode, artwork_url, prompt }); } catch (err) { return json({ success: false, error: String(err?.message || err) }, 500); } }); function buildArtworkPrompt(m: any) { return ` Create an original dramatic oil painting symbolizing this sports mistake. Sports mistake: Title: ${m.ai_generated_title || m.title || ""} Sport/League: ${m.ai_generated_sport || m.league || m.sport || ""} Year: ${m.year || ""} Category: ${m.ai_generated_category || m.category || ""} Teams/People: ${m.ai_generated_teams_people || m.teams_people || ""} Summary: ${m.ai_generated_summary || m.summary || ""} Why it matters: ${m.ai_generated_why_it_matters || m.why_it_matters || ""} Style: Editorial sports-history oil painting. Cinematic lighting. Painterly brush strokes. Emotional, tense, dramatic. Strong thumbnail composition. Generic uniforms and generic sports setting. Important restrictions: Do not include any readable letters, logos, initials, team marks, cap letters, jersey text, scoreboard text, signage, or symbols. Do not include trademarked team logos. Do not copy a broadcast frame or known photograph. Do not depict exact real-person likenesses. Avoid all team-identifying marks. Focus on emotion and symbolism, not documentary accuracy. `; } function json(data: unknown, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...corsHeaders, "Content-Type": "application/json" }, }); }
